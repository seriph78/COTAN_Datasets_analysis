reOrder <- function(df, numBlocks) {
  blockLength <- nrow(df) / numBlocks
  permut <- rep(1:blockLength, each = numBlocks) +
              rep(seq(0, nrow(df) - 1, by = blockLength), times = numBlocks)
  return(df[permut, ])
}

mg <- function(mixings) {
  res <- mixings
  res[res !=0 ] <- ceiling(log2(round(res[res !=0 ] * 40)))
  return(res)
}
