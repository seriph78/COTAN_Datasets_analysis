  p_values <- numeric(ncol(subset_data))

    if (i == j) {
      p_values[j] <- 1
    } else {
      cor_result <- cor.test(subset_data[, i], subset_data[, j], method = "pearson")
      p_values[j] <- cor_result$p.value
    }
  }
  return(p_values)
}
